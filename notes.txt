// save_notifications.js
$(function () {
    if (!window.userId) return;

    // Формируем WebSocket URL
    const protocol = window.location.protocol === "https:" ? "wss" : "ws";
    const wsUrl = `${protocol}://${window.location.host}/ws/notifications/${window.userId}/`;

    const socket = new WebSocket(wsUrl);

    socket.onopen = function () {
        console.log("WebSocket connected for user " + window.userId);
    };

    socket.onmessage = function (event) {
        const data = JSON.parse(event.data);
        if (!data.type) return;

        switch (data.type) {
            case "start":
                toastr.info("Начинаем сохранение слов...", "Сохранение", {
                    timeOut: 0,
                    extendedTimeOut: 0,
                    closeButton: true,
                    tapToDismiss: false
                });
                break;

            case "progress":
                toastr.clear();
                toastr.info(`Сохранено слов: ${data.current} / ${data.total} (${data.percent}%)`, "Сохранение", {
                    timeOut: 0,
                    extendedTimeOut: 0,
                    closeButton: true,
                    tapToDismiss: false
                });
                break;

            case "done":
                toastr.clear();
                toastr.success("Слова успешно сохранены!", "Сохранение", {timeOut: 3000});
                break;

            case "error":
                toastr.clear();
                toastr.error(`Ошибка: ${data.message}`, "Сохранение", {timeOut: 5000});
                break;

            default:
                console.warn("Unknown WS type:", data.type);
        }
    };

    socket.onclose = function () {
        console.log("WebSocket disconnected for user " + window.userId);
    };

    socket.onerror = function (err) {
        console.error("WebSocket error:", err);
    };
});

import os

os.environ.setdefault("DJANGO_SETTINGS_MODULE", "config.settings")

from django.core.asgi import get_asgi_application
from channels.routing import ProtocolTypeRouter, URLRouter
from channels.auth import AuthMiddlewareStack

import config.routing

django_asgi_app = get_asgi_application()
application = ProtocolTypeRouter(
    {
        "http": django_asgi_app,
        "websocket": AuthMiddlewareStack(
            URLRouter(config.routing.websocket_urlpatterns)
        ),
    }
)
